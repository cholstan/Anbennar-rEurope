	desc = tcharialt.1.d
	picture = EFFELAI_BLOOM_ANIMATED_eventPicture

    option = { #stay elven forebear
        name = tcharialt.1.a 
        ai_chance = { factor = 100 }
        trigger = {

            duration = 9150
        }
    }
    option = { #go fangaulan
        name = tcharialt.1.b

            always = yes

        change_religion = fangaulan_pantheon 
        add_historical_friend = L22
        L22 = {
            add_historical_friend = ROOT
            create_marriage = ROOT
        }
        add_prestige = 100
    }
}
