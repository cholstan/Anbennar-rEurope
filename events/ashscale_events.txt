country_event = { #A new Identity
	id = ash_flavour_events.8
	title = ash_flavour_events.8.t
	desc = ash_flavour_events.8.d
	picture = KOBOLD_HOARD_eventPicture

	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = ash_flavour_events.8.a
		ROOT = {
			change_tag = EC3
			country_event = { id = ideagroups.1 } #Swap Ideas
			hidden_effect = {
				change_government = monarchy
				set_government_rank = 2
			}
			add_government_reform = techno_monarchy_reform
			add_country_modifier = {
				name = "centralization_modifier"
				duration = 7300
			}
			hidden_effect = {
				every_owned_province = {
					limit = {
						culture_group = kobold
					}
					change_culture = darkscale_kobold
				}
			}
			if = {
				limit = { 
					AND = {
						ruler_culture = ROOT
						NOT = {
							ruler_culture = darkscale_kobold
						}
					}
				}
				set_ruler_culture = darkscale_kobold
			}
			if = {
				limit = { has_heir = yes heir_culture = ROOT }
				set_heir_culture = darkscale_kobold
			}
			if = {
				limit = { has_consort = yes consort_culture = ROOT }
				set_consort_culture = darkscale_kobold
			}
			change_primary_culture = darkscale_kobold
		}
	}
}
